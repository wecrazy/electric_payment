// Establish WebSocket connection
let wsProtocolLockRow = window.location.protocol === "https:" ? "wss:" : "ws:";
let socketUrlLockRow = `${wsProtocolLockRow}//${window.location.host}/ws-lock`;
const socketLockRow = new WebSocket(socketUrlLockRow);

const userID = "{{.userid}}"; // Replace dynamically
let lockedTaskId = null; // Store locked row ID

// **Handle WebSocket messages (lock/unlock rows)**
socketLockRow.onmessage = (event) => {
    const data = JSON.parse(event.data);
    console.log("WebSocket Message:", data);

    let row = $(`#DataTables_Table_0 tbody tr[data-row-id='${data.row_id}']`);
    let detailButton = row.find(".detail-btn");

    if (data.action === "lock") {
        if (data.user_id !== userID) {
            row.addClass("locked").css("pointer-events", "none");
            detailButton.prop("disabled", true).css("cursor", "not-allowed");
        } else {
            row.addClass("locked").css("background-color", "#ffcccc");
        }
    } else if (data.action === "unlock") {
        row.removeClass("locked").css({ "pointer-events": "auto", "background-color": "" });
        detailButton.prop("disabled", false).css("cursor", "pointer");
    }
};

// **Prevent opening modal on locked row**
$('#DataTables_Table_0 tbody').on('click', '.detail-btn', function (event) {
    let row = $(this).closest('tr');
    let rowID = row.attr('data-row-id');

    if (row.hasClass("locked")) {
        alert("This row is locked by another user.");
        event.preventDefault(); // Stop modal from opening
        return false;
    }

    // Send lock request if row is not locked
    socketLockRow.send(JSON.stringify({
        action: "lock",
        row_id: rowID,
        user_id: userID
    }));

    lockedTaskId = rowID; // Save locked row ID
});

// **Observe modal opening and lock row**
const observer = new MutationObserver((mutations) => {
    mutations.forEach((mutation) => {
        mutation.addedNodes.forEach((node) => {
            if ($(node).hasClass("dtr-bs-modal")) {
                let rowID = $(node).find("td:contains('I D Task:') + td p").text().trim();
                if (!rowID) return;

                lockedTaskId = rowID;
                console.log("Locking row:", rowID);

                // Send lock request
                socketLockRow.send(JSON.stringify({
                    action: "lock",
                    row_id: rowID,
                    user_id: userID
                }));

                // Disable button for other users
                let row = $(`#DataTables_Table_0 tbody tr[data-row-id='${rowID}']`);
                row.find(".detail-btn").prop("disabled", true).css("cursor", "not-allowed");
            }
        });
    });
});
observer.observe(document.body, { childList: true, subtree: true });

// **Detect modal close and unlock the row**
$(document).on("click", "[data-bs-dismiss='modal']", function () {
    if (!lockedTaskId) return;

    console.log("Unlocking row:", lockedTaskId);
    socketLockRow.send(JSON.stringify({
        action: "unlock",
        row_id: lockedTaskId,
        user_id: userID
    }));

    let row = $(`#DataTables_Table_0 tbody tr[data-row-id='${lockedTaskId}']`);
    row.find(".detail-btn").prop("disabled", false).css("cursor", "pointer");

    lockedTaskId = null;
});

        $("button.dt-button.btn-label-dark").each(function () {
            if ($(this).find("i.bx-expand-vertical").length) {
